IPBLOCK | CREATE TABLE `IPBLOCK` (
  `IP` varchar(16) NOT NULL,
  `CREATIONTIME` timestamp NOT NULL DEFAULT current_timestamp(),
  `CATEGORY` tinyint(3) unsigned DEFAULT NULL,
  PRIMARY KEY (`IP`),
  KEY `CATEGORY` (`CATEGORY`),
  CONSTRAINT `IPBLOCK_ibfk_1` FOREIGN KEY (`CATEGORY`) REFERENCES `URLCAT` (`CATEGORY`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 

URLCAT | CREATE TABLE `URLCAT` (
  `CATEGORY` tinyint(3) unsigned NOT NULL AUTO_INCREMENT,
  `CREATIONTIME` timestamp NOT NULL DEFAULT current_timestamp(),
  `URL` varchar(256) NOT NULL,
  `CATEGORYNAME` varchar(64) NOT NULL,
  PRIMARY KEY (`CATEGORY`),
  UNIQUE KEY `URL` (`URL`)
) ENGINE=InnoDB AUTO_INCREMENT=4 DEFAULT CHARSET=utf8
